CREATE TABLE `tccecossistemaescolar`.`escola` (
  `EscolaGUID` CHAR(36) NOT NULL,
  `EscolaNome` VARCHAR(100) NULL,
  `EscolaCorPriEs` CHAR(6) NULL DEFAULT '000000',
  `EscolaCorPriCl` CHAR(6) NULL DEFAULT '000000',
  `EscolaCorSecEs` CHAR(6) NULL DEFAULT '000000',
  `EscolaCorSecCl` CHAR(6) NULL DEFAULT '000000',
  `EscolaIcone` BLOB NULL,
  PRIMARY KEY (`EscolaGUID`)
);

CREATE TABLE `tccecossistemaescolar`.`usuario` (
  `UsuarioCPF` VARCHAR(14) NOT NULL,
  `UsuarioEmail` VARCHAR(60) NULL,
  `UsuarioId` VARCHAR(45) NULL,
  `UsuarioTelefone` VARCHAR(15) NULL,
  `UsuarioNome` VARCHAR(100) NOT NULL,
  `UsuarioSenha` VARCHAR(100) NOT NULL,
  PRIMARY KEY (`UsuarioCPF`)
);

CREATE TABLE `tccecossistemaescolar`.`funcao` (
  `FuncaoId` INT NOT NULL,
  `FuncaoNome` VARCHAR(30) NOT NULL,
  PRIMARY KEY (`FuncaoId`)
);

CREATE TABLE `tccecossistemaescolar`.`escolaxusuarioxfuncao` (
  `EscolaxUsuarioxFuncaoId` INT NOT NULL AUTO_INCREMENT,
  `UsuarioCPF` VARCHAR(14) NOT NULL,
  `EscolaGUID` CHAR(36) NOT NULL,
  `FuncaoId` INT NOT NULL,
  PRIMARY KEY (`EscolaxUsuarioxFuncaoId`),
  UNIQUE KEY `UQ_EscolaUsuarioFuncao` (`UsuarioCPF`, `EscolaGUID`, `FuncaoId`),
  CONSTRAINT `FK_EUF_Usuario` FOREIGN KEY (`UsuarioCPF`)
    REFERENCES `tccecossistemaescolar`.`usuario` (`UsuarioCPF`)
    ON UPDATE CASCADE
    ON DELETE RESTRICT,
  CONSTRAINT `FK_EUF_Escola` FOREIGN KEY (`EscolaGUID`)
    REFERENCES `tccecossistemaescolar`.`escola` (`EscolaGUID`)
    ON UPDATE CASCADE
    ON DELETE RESTRICT,
  CONSTRAINT `FK_EUF_Funcao` FOREIGN KEY (`FuncaoId`)
    REFERENCES `tccecossistemaescolar`.`funcao` (`FuncaoId`)
    ON UPDATE CASCADE
    ON DELETE RESTRICT
);
